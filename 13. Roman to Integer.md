class Solution:
    def romanToInt(self, s: str) -> int:
        sym_val = {
        'I': 1,
        'V': 5,
        'X': 10,
        'L': 50,
        'C': 100,
        'D': 500,
        'M': 1000,
    }
        res = 0
        for i, ch in enumerate(s):
            value = sym_val[ch]
            if i < len(s) - 1 and value < sym_val[s[i + 1]]:#最后一个不用比较
                res -= value
            else: 
                res += value
        return res

``
