```python
# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def addTwoNumbers(self, l1: Optional[ListNode], l2: Optional[ListNode]) -> Optional[ListNode]:
        dummy = ListNode(-1)
        cur = dummy
        carry = 0
        
        while l1 or l2:
            if l1:
                x = l1.val
            else:
                x = 0
            # x = l1.val if l1 else 0   
            if l2:
                y  = l2.val
            else:
                y = 0
            total = x + y + carry
            carry = total // 10
            cur.next = ListNode(total % 10)
            cur = cur.next
            
            if l1:
                l1 = l1.next
            if l2:
                l2 = l2.next
                
        if carry:
            cur.next = ListNode(carry)
            
        return dummy.next
                
        
      

```
