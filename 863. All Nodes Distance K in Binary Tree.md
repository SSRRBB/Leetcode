## 题目：

https://leetcode.com/problems/all-nodes-distance-k-in-binary-tree/


## 思路：

DFS去构建无向图

BFS去遍历图


## 答案：
```python
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def distanceK(self, root: TreeNode, target: TreeNode, k: int) -> List[int]:
        #构建图(dfs) +BFS遍历图
         
       # 巧妙之处，有parent才去构建连通关系
      # 注意这里需要传一个parent node，否则数字无法判断是否None
        graph = collections.defaultdict(list)
        def dfs(root, par):
            if not root:
                return
            if par:
                graph[par.val].append(root.val)
                graph[root.val].append(par.val)
            dfs(root.left, root)
            dfs(root.right, root)
            
        dfs(root, None)
        ##print(graph)
        ##{3: [5, 1], 5: [3, 6, 2], 6: [5], 2: [5, 7, 4], 7: [2], 4: [2], 1: [3, 0, 8], 0: [1], 8: [1]})

        # BFS 的经典遍历图方法
        ans = []
        visited = set([target.val]) #一定要有[]
        #visited = {target.val}#这样也是set
        queue = collections.deque([target.val])

        while queue:
            queue_len = len(queue)
           # print(queue) deque([5]);deque([3,6,2]);deque([1,7,4]);deque([5])
           # print(visited) {5} {2, 3, 5, 6} {1, 2, 3, 4, 5, 6, 7}
            # 当 k 为0 则输出
            if k == 0:
                for _ in range(queue_len):
                    curr = queue.popleft()
                    ans.append(curr)
                return ans
            
            for _ in range(queue_len):
                curr = queue.popleft()
                for n in graph[curr]:
                    if n not in visited:
                        queue.append(n)
                        visited.add(n)
            k -= 1
        
        return ans           
        
```
