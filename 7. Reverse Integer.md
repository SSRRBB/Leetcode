**str**
```python
class Solution:
    def reverse(self, x: int) -> int:
    
        if x > 2**31 - 1 or x < -2**31:
            return 0
       
        def rev(x):
            s = str(x)
            res = s[::-1]
            if int(res) > 2**31 - 1 or -int(res)  < -2**31:
                return 0 ## 这一步不能忘啊！！
            return int(res)
            
        if x > 0:
            return rev(x)
        else:
            return -rev(-x)
```
**math**
```python
class Solution:
    def reverse(self, x: int) -> int:
        ##math
        if x > 2**31 - 1 or x < -2**31:
            return 0
        
        cur = 0
        y  = abs(x)
        while y != 0:
            cur = cur * 10 + y % 10 # y % 10每次取末尾数字
            y //= 10
            
        if cur > 2**31 - 1 or -cur < -2**31:
            return 0
            
        return cur if x > 0 else -cur
```
