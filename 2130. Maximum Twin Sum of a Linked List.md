**list + two pointer**
```python
# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def pairSum(self, head: Optional[ListNode]) -> int:
        #先变成数组，再双指针
        p = head
        tmp = []
        while p:
            tmp.append(p.val)
            p = p.next
        
        res = float('-inf')
        i = 0
        j = len(tmp) - 1
        while i < j:
            res = max(res, tmp[i] + tmp[j])
            i += 1
            j -= 1
        return res
```

# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next

    def pairSum(self, head: Optional[ListNode]) -> int:
    ######## 快慢指针+反转
        res = 0
        # 快慢指针，找到链表的中间节点(876) [5,4,2,1]
        dummy = ListNode(-1)
        dummy.next = head
        
        fast, slow = dummy, dummy
        
        while fast.next and fast.next.next:
            fast = fast.next.next
            slow = slow.next #slow = 4, fast = 1
        slow = slow.next #slow = 2
        # (206)根据孪生节点的定义， 第 i 个节点与第 (n-1-i) 个节点 为孪生节点，需要反转第二部分的链表的节点
        pre = None
        cur = slow
        while cur:
            temp = cur.next # 先把原来cur.next位置存起来
            cur.next = pre # 接着把 cur.next 指向前驱节点 pre
            pre = cur # pre节点前移
            cur = temp # cur节点前移
        #变成【5，4】和【2，1】
        # 配对孪生节点找到孪生节点的最大和
        while pre:
            res = max(res, head.val + pre.val)
            pre, head = pre.next, head.next

        return res
